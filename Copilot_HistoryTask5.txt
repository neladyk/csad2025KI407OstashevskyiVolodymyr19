name: CI Cross Build and Test

on:
  push:
    branches:
      - '**develop**'
      - main
  pull_request:
    branches:
      - '**develop**'
      - main

jobs:
  cross_build:
    runs-on: ${{ matrix.os }}
    strategy:
      matrix:
        os: [ubuntu-latest, windows-latest, macos-latest]

    steps:
      - name: Checkout code
        uses: actions/checkout@v4

      - name: Install Build Dependencies

          sudo apt-get update
